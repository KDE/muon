    main.cpp
    MuonDiscoverMainWindow.cpp

    DiscoverAction.cpp
    NativeScrollBar.cpp
    MuonActionGroup.cpp
    PaginateModel.cpp

                                     ${QT_QTOPENGL_LIBRARY}
                                     ${QT_QTDECLARATIVE_LIBRARY}
)
